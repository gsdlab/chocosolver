001package org.clafer.ir;
002
003/**
004 * An immutable expression that evaluates to a set of integers.
005 *
006 * @author jimmy
007 */
008public interface IrSetExpr extends IrSet, IrExpr {
009
010    /**
011     * Dynamic dispatch on the visitor.
012     *
013     * @param <A> the parameter type
014     * @param <B> the return type
015     * @param visitor the visitor
016     * @param a the parameter
017     * @return the return value
018     */
019    public <A, B> B accept(IrSetExprVisitor<A, B> visitor, A a);
020}

001package org.clafer.ir;
002
003import gnu.trove.iterator.TIntIterator;
004import org.clafer.collection.BoundIntIterator;
005import org.clafer.collection.ReverseBoundIntIterator;
006
007/**
008 * A contiguous domain between a low and high bound.
009 *
010 * @author jimmy
011 */
012public class IrBoundDomain implements IrDomain {
013
014    private final int low;
015    private final int high;
016
017    /**
018     * @param low lowest value in the domain, inclusive
019     * @param high highest value in the domain, inclusive
020     */
021    public IrBoundDomain(int low, int high) {
022        if (low > high) {
023            throw new IllegalArgumentException();
024        }
025        this.low = low;
026        this.high = high;
027    }
028
029    /**
030     * {@inheritDoc}
031     */
032    @Override
033    public boolean isBounded() {
034        return true;
035    }
036
037    /**
038     * {@inheritDoc}
039     */
040    @Override
041    public boolean contains(int value) {
042        return value >= low && value <= high;
043    }
044
045    /**
046     * {@inheritDoc}
047     */
048    @Override
049    public int getLowBound() {
050        return low;
051    }
052
053    /**
054     * {@inheritDoc}
055     */
056    @Override
057    public int getHighBound() {
058        return high;
059    }
060
061    /**
062     * {@inheritDoc}
063     */
064    @Override
065    public boolean isEmpty() {
066        return false;
067    }
068
069    /**
070     * {@inheritDoc}
071     */
072    @Override
073    public int size() {
074        return high + 1 - low;
075    }
076
077    /**
078     * {@inheritDoc}
079     */
080    @Override
081    public int[] getValues() {
082        int[] values = new int[size()];
083        for (int i = 0; i < values.length; i++) {
084            values[i] = low + i;
085        }
086        return values;
087    }
088
089    /**
090     * {@inheritDoc}
091     */
092    @Override
093    public TIntIterator iterator() {
094        return iterator(true);
095    }
096
097    /**
098     * {@inheritDoc}
099     */
100    @Override
101    public TIntIterator iterator(boolean increasing) {
102        return increasing ? new BoundIntIterator(low, high) : new ReverseBoundIntIterator(low, high);
103    }
104
105    /**
106     * {@inheritDoc}
107     */
108    @Override
109    public boolean equals(Object obj) {
110        if (this == obj) {
111            return true;
112        }
113        if (obj instanceof IrDomain) {
114            IrDomain other = (IrDomain) obj;
115            if (other.isEmpty()) {
116                return false;
117            }
118            if (other.isBounded()) {
119                return low == other.getLowBound() && high == other.getHighBound();
120            }
121            int[] otherValues = other.getValues();
122            if (size() != other.size()) {
123                return false;
124            }
125            for (int i = 0; i < otherValues.length; i++) {
126                if (otherValues[i] != low + i) {
127                    return false;
128                }
129            }
130            return true;
131        }
132        return false;
133    }
134
135    /**
136     * {@inheritDoc}
137     */
138    @Override
139    public int hashCode() {
140        return low ^ high;
141    }
142
143    /**
144     * {@inheritDoc}
145     */
146    @Override
147    public String toString() {
148        return "{" + low + ", ..., " + high + "}";
149    }
150}

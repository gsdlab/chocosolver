001package org.clafer.ir.compiler;
002
003import java.util.ArrayList;
004import java.util.List;
005import java.util.Map;
006import java.util.Map.Entry;
007import org.clafer.ir.IrBoolConstant;
008import org.clafer.ir.IrBoolVar;
009import org.clafer.ir.IrIntConstant;
010import org.clafer.ir.IrIntVar;
011import org.clafer.ir.IrSetConstant;
012import org.clafer.ir.IrSetVar;
013import org.clafer.ir.IrUtil;
014import solver.variables.BoolVar;
015import solver.variables.IntVar;
016import solver.variables.SetVar;
017
018/**
019 * Maps IR non-constant variables to their translated Choco variables. The
020 * get&lt;Type&gt;Var methods fail on constants. The get&lt;Type&gt;Value
021 * methods handles constants, and is the preferred way of lookup.
022 *
023 * @author jimmy
024 */
025public class IrSolutionMap {
026
027    private final Map<IrBoolVar, BoolVar> boolVars;
028    private final Map<IrIntVar, IrIntVar> coalescedIntVars;
029    private final Map<IrIntVar, IntVar> intVars;
030    private final Map<IrSetVar, SetVar> setVars;
031
032    IrSolutionMap(Map<IrBoolVar, BoolVar> boolVars,
033            Map<IrIntVar, IrIntVar> coalescedIntVars,
034            Map<IrIntVar, IntVar> intVars,
035            Map<IrSetVar, SetVar> setVars) {
036        this.boolVars = boolVars;
037        this.coalescedIntVars = coalescedIntVars;
038        this.intVars = intVars;
039        this.setVars = setVars;
040    }
041
042    public BoolVar getBoolVar(IrBoolVar var) {
043        return IrUtil.notNull("Bool var " + var + " not par of IR solution", boolVars.get(var));
044    }
045
046    public boolean getBoolValue(IrBoolVar var) {
047        if (var instanceof IrBoolConstant) {
048            return ((IrBoolConstant) var).getValue();
049        }
050        return getBoolVar(var).getValue() != 0;
051    }
052
053    public BoolVar[] getBoolVars(IrBoolVar... vars) {
054        BoolVar[] bvars = new BoolVar[vars.length];
055        for (int i = 0; i < bvars.length; i++) {
056            bvars[i] = getBoolVar(vars[i]);
057        }
058        return bvars;
059    }
060
061    public boolean[] getBoolValues(IrBoolVar... vars) {
062        boolean[] bvalues = new boolean[vars.length];
063        for (int i = 0; i < bvalues.length; i++) {
064            bvalues[i] = getBoolValue(vars[i]);
065        }
066        return bvalues;
067    }
068
069    public BoolVar[] getBoolVars() {
070        return boolVars.values().toArray(new BoolVar[boolVars.size()]);
071    }
072
073    public BoolVar[] getBoolDecisionVars() {
074        List<BoolVar> decisionVars = new ArrayList<BoolVar>(boolVars.size());
075        for (Entry<IrBoolVar, BoolVar> entry : boolVars.entrySet()) {
076            if (entry.getKey().isDecision()) {
077                decisionVars.add(entry.getValue());
078            }
079        }
080        return decisionVars.toArray(new BoolVar[decisionVars.size()]);
081    }
082
083    public IntVar getIntVar(IrIntVar var) {
084        IrIntVar intVar = coalescedIntVars.get(var);
085        if (intVar == null) {
086            intVar = var;
087        }
088        return IrUtil.notNull("Int var " + var + " not par of IR solution", intVars.get(intVar));
089    }
090
091    public int getIntValue(IrIntVar var) {
092        if (var instanceof IrIntConstant) {
093            return ((IrIntConstant) var).getValue();
094        }
095        return getIntVar(var).getValue();
096    }
097
098    public IntVar[] getIntVars(IrIntVar... vars) {
099        IntVar[] ivars = new IntVar[vars.length];
100        for (int i = 0; i < ivars.length; i++) {
101            ivars[i] = getIntVar(vars[i]);
102        }
103        return ivars;
104    }
105
106    public int[] getIntValues(IrIntVar... vars) {
107        int[] ivalues = new int[vars.length];
108        for (int i = 0; i < ivalues.length; i++) {
109            ivalues[i] = getIntValue(vars[i]);
110        }
111        return ivalues;
112    }
113
114    public IntVar[] getIntVars() {
115        return intVars.values().toArray(new IntVar[intVars.size()]);
116    }
117
118    public IntVar[] getIntDecisionVars() {
119        List<IntVar> decisionVars = new ArrayList<IntVar>(intVars.size());
120        for (Entry<IrIntVar, IntVar> entry : intVars.entrySet()) {
121            if (entry.getKey().isDecision()) {
122                decisionVars.add(entry.getValue());
123            }
124        }
125        return decisionVars.toArray(new IntVar[decisionVars.size()]);
126    }
127
128    public SetVar getSetVar(IrSetVar var) {
129        return IrUtil.notNull("Set var " + var + " not par of IR solution", setVars.get(var));
130    }
131
132    public int[] getSetValue(IrSetVar var) {
133        if (var instanceof IrSetConstant) {
134            return ((IrSetConstant) var).getValue();
135        }
136        return getSetVar(var).getValue();
137    }
138
139    public SetVar[] getSetVars(IrSetVar... vars) {
140        SetVar[] svars = new SetVar[vars.length];
141        for (int i = 0; i < svars.length; i++) {
142            svars[i] = getSetVar(vars[i]);
143        }
144        return svars;
145    }
146
147    public int[][] getSetValues(IrSetVar... vars) {
148        int[][] svalues = new int[vars.length][];
149        for (int i = 0; i < svalues.length; i++) {
150            svalues[i] = getSetValue(vars[i]);
151        }
152        return svalues;
153    }
154
155    public SetVar[] getSetVars() {
156        return setVars.values().toArray(new SetVar[setVars.size()]);
157    }
158
159    public SetVar[] getSetDecisionVars() {
160        List<SetVar> decisionVars = new ArrayList<SetVar>(setVars.size());
161        for (Entry<IrSetVar, SetVar> entry : setVars.entrySet()) {
162            if (entry.getKey().isDecision()) {
163                decisionVars.add(entry.getValue());
164            }
165        }
166        return decisionVars.toArray(new SetVar[decisionVars.size()]);
167    }
168}

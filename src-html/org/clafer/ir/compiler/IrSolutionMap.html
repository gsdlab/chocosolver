001package org.clafer.ir.compiler;
002
003import org.clafer.collection.ReadMap;
004import org.clafer.ir.IrBoolConstant;
005import org.clafer.ir.IrBoolVar;
006import org.clafer.ir.IrIntConstant;
007import org.clafer.ir.IrIntVar;
008import org.clafer.ir.IrSetConstant;
009import org.clafer.ir.IrSetVar;
010import org.clafer.ir.IrUtil;
011import solver.variables.BoolVar;
012import solver.variables.IntVar;
013import solver.variables.SetVar;
014
015/**
016 * Maps IR non-constant variables to their translated Choco variables. The
017 * get&lt;Type&gt;Var methods fail on constants. The get&lt;Type&gt;Value
018 * methods handles constants, and is the preferred way of lookup.
019 *
020 * @author jimmy
021 */
022public class IrSolutionMap {
023
024    private final ReadMap<IrBoolVar, BoolVar> boolVars;
025    private final ReadMap<IrIntVar, IntVar> intVars;
026    private final ReadMap<IrSetVar, SetVar> setVars;
027
028    IrSolutionMap(ReadMap<IrBoolVar, BoolVar> boolVars, ReadMap<IrIntVar, IntVar> intVars, ReadMap<IrSetVar, SetVar> setVars) {
029        this.boolVars = boolVars.readOnly();
030        this.intVars = intVars.readOnly();
031        this.setVars = setVars.readOnly();
032    }
033
034    public BoolVar getBoolVar(IrBoolVar var) {
035        return IrUtil.notNull("Bool var " + var + " not par of IR solution", boolVars.get(var));
036    }
037
038    public boolean getBoolValue(IrBoolVar var) {
039        if (var instanceof IrBoolConstant) {
040            return ((IrBoolConstant) var).getValue();
041        }
042        return getBoolVar(var).getValue() != 0;
043    }
044
045    public BoolVar[] getBoolVars(IrBoolVar... vars) {
046        BoolVar[] bvars = new BoolVar[vars.length];
047        for (int i = 0; i < bvars.length; i++) {
048            bvars[i] = getBoolVar(vars[i]);
049        }
050        return bvars;
051    }
052
053    public boolean[] getBoolValues(IrBoolVar... vars) {
054        boolean[] bvalues = new boolean[vars.length];
055        for (int i = 0; i < bvalues.length; i++) {
056            bvalues[i] = getBoolValue(vars[i]);
057        }
058        return bvalues;
059    }
060
061    public BoolVar[] getBoolVars() {
062        return boolVars.getValues().toArray(new BoolVar[boolVars.size()]);
063    }
064
065    public IntVar getIntVar(IrIntVar var) {
066        return IrUtil.notNull("Int var " + var + " not par of IR solution", intVars.get(var));
067    }
068
069    public int getIntValue(IrIntVar var) {
070        if (var instanceof IrIntConstant) {
071            return ((IrIntConstant) var).getValue();
072        }
073        return getIntVar(var).getValue();
074    }
075
076    public IntVar[] getIntVars(IrIntVar... vars) {
077        IntVar[] ivars = new IntVar[vars.length];
078        for (int i = 0; i < ivars.length; i++) {
079            ivars[i] = getIntVar(vars[i]);
080        }
081        return ivars;
082    }
083
084    public int[] getIntValues(IrIntVar... vars) {
085        int[] ivalues = new int[vars.length];
086        for (int i = 0; i < ivalues.length; i++) {
087            ivalues[i] = getIntValue(vars[i]);
088        }
089        return ivalues;
090    }
091
092    public IntVar[] getIntVars() {
093        return intVars.getValues().toArray(new IntVar[intVars.size()]);
094    }
095
096    public SetVar getSetVar(IrSetVar var) {
097        return IrUtil.notNull("Set var " + var + " not par of IR solution", setVars.get(var));
098    }
099
100    public int[] getSetValue(IrSetVar var) {
101        if (var instanceof IrSetConstant) {
102            return ((IrSetConstant) var).getValue();
103        }
104        return getSetVar(var).getValue();
105    }
106
107    public SetVar[] getSetVars(IrSetVar... vars) {
108        SetVar[] svars = new SetVar[vars.length];
109        for (int i = 0; i < svars.length; i++) {
110            svars[i] = getSetVar(vars[i]);
111        }
112        return svars;
113    }
114
115    public int[][] getSetValues(IrSetVar... vars) {
116        int[][] svalues = new int[vars.length][];
117        for (int i = 0; i < svalues.length; i++) {
118            svalues[i] = getSetValue(vars[i]);
119        }
120        return svalues;
121    }
122
123    public SetVar[] getSetVars() {
124        return setVars.getValues().toArray(new SetVar[setVars.size()]);
125    }
126}

028    /**
029     * Optimize the module.
030     *
031     * @param module the module to optimize
032     * @return the optimized module
033     */
034    public static IrModule optimize(IrModule module) {
035        return module.withConstraints(optimizer.rewrite(and(module.getConstraints()), null));
036    }
037    private static final IrRewriter<Void> optimizer = new IrRewriter<Void>() {
038        @Override
039        public IrBoolExpr visit(IrLone ir, Void a) {
040            IrBoolExpr[] operands = rewrite(ir.getOperands(), a);
041            if (operands.length == 2) {
042                if (operands[0] instanceof IrCompare) {
043                    IrBoolExpr antecedent = operands[1];
044                    IrCompare compare = (IrCompare) operands[0];
045                    IrBoolExpr opt = optimizeLoneCompare(antecedent, compare.getLeft(), compare.getOp(), compare.getRight());
046                    if (opt == null) {
047                        opt = optimizeLoneCompare(antecedent, compare.getRight(), compare.getOp().reverse(), compare.getLeft());
048                    }
049                    if (opt != null) {
050                        return opt;
051                    }
052                }
053                if (operands[1] instanceof IrCompare) {
054                    IrBoolExpr antecedent = operands[0];
055                    IrCompare compare = (IrCompare) operands[1];
056                    IrBoolExpr opt = optimizeLoneCompare(antecedent, compare.getLeft(), compare.getOp(), compare.getRight());
057                    if (opt == null) {
058                        opt = optimizeLoneCompare(antecedent, compare.getRight(), compare.getOp().reverse(), compare.getLeft());
059                    }
060                    if (opt != null) {
061                        return opt;
062                    }
063                }
064            }
065            return lone(operands);
066        }
067
068        @Override
069        public IrBoolExpr visit(IrOr ir, Void a) {
070            IrBoolExpr[] operands = rewrite(ir.getOperands(), a);
071            if (operands.length == 2) {
072                if (operands[0] instanceof IrCompare) {
073                    IrBoolExpr antecedent = operands[1];
074                    IrCompare compare = (IrCompare) operands[0];
075                    IrBoolExpr opt = optimizeOrCompare(antecedent, compare.getLeft(), compare.getOp(), compare.getRight());
076                    if (opt == null) {
077                        opt = optimizeOrCompare(antecedent, compare.getRight(), compare.getOp().reverse(), compare.getLeft());
078                    }
079                    if (opt != null) {
080                        return opt;
081                    }
082                }
083                if (operands[1] instanceof IrCompare) {
084                    IrBoolExpr antecedent = operands[0];
085                    IrCompare compare = (IrCompare) operands[1];
086                    IrBoolExpr opt = optimizeOrCompare(antecedent, compare.getLeft(), compare.getOp(), compare.getRight());
087                    if (opt == null) {
088                        opt = optimizeOrCompare(antecedent, compare.getRight(), compare.getOp().reverse(), compare.getLeft());
089                    }
090                    if (opt != null) {
091                        return opt;
092                    }
093                }
094            }
095            return or(operands);
096        }
097
098        @Override
099        public IrBoolExpr visit(IrImplies ir, Void a) {
100            // Rewrite
101            //     !a => !b
102            // to
103            //     b => a
104            if (ir.getAntecedent().isNegative() && ir.getConsequent().isNegative()) {
105                return rewrite(implies(not(ir.getConsequent()), not(ir.getAntecedent())), a);
106            }
107            // Rewrite
108            //     !a => b
109            // to
110            //     a or b
111            if (ir.getAntecedent().isNegative()) {
112                return rewrite(or(not(ir.getAntecedent()), ir.getConsequent()), a);
113            }
114            // Rewrite
115            //     a => !b
116            // to
117            //     a + b <= 1
118            if (ir.getConsequent().isNegative()) {
119                return rewrite(lone(ir.getAntecedent(), not(ir.getConsequent())), a);
120            }
121            IrBoolExpr antecedent = rewrite(ir.getAntecedent(), a);
122            IrBoolExpr consequent = rewrite(ir.getConsequent(), a);
123            if (consequent instanceof IrCompare) {
124                IrCompare compare = (IrCompare) consequent;
125                IrBoolExpr opt = optimizeImplicationCompare(antecedent, compare.getLeft(), compare.getOp(), compare.getRight());
126                if (opt == null) {
127                    opt = optimizeImplicationCompare(antecedent, compare.getRight(), compare.getOp().reverse(), compare.getLeft());
128                }
129                if (opt != null) {
130                    return opt;
131                }
132            }
133            return implies(antecedent, consequent);
134        }
135
136        @Override
137        public IrSetExpr visit(IrOffset ir, Void a) {
138            if (ir.getSet() instanceof IrOffset) {
139                // Rewrite
140                //    offset(offset(set, a), b)
141                // to
142                //    offset(set, a + b)
143                // This optimization is important for going multiple steps up the
144                // hierarchy.
145                IrOffset innerOffset = (IrOffset) ir.getSet();
146                return rewrite(offset(innerOffset.getSet(),
147                        ir.getOffset() + innerOffset.getOffset()), a);
148            }
149            return super.visit(ir, a);
150        }
151    };

153    /**
154     * Optimize {@code lone(antecedent, left `op` right)} where `op` is = or !=.
155     */
156    private static IrBoolExpr optimizeLoneCompare(IrBoolExpr antecedent, IrIntExpr left, IrCompare.Op op, IrIntExpr right) {
157        IrDomain domain = left.getDomain();
158        Integer constant = IrUtil.getConstant(right);
159        if (domain.size() == 2 && constant != null) {
160            switch (op) {
161                case Equal:
162                    // Rewrite
163                    //     lone(bool, int = 888)
164                    //         where dom(int) = {-3, 888}
165                    // to
166                    //     asInt(bool) <= 888 - int
167                    //     asInt(bool) + int <= 888
168                    if (domain.getHighBound() == constant.intValue()) {
169                        return lessThanEqual(add(asInt(antecedent), left),
170                                domain.getHighBound());
171                    }
172                    // Rewrite
173                    //     lone(bool, int = -3)
174                    //         where dom(int) = {-3, 888}
175                    // to
176                    //     asInt(bool) <= int - (-3)
177                    if (domain.getLowBound() == constant.intValue()) {
178                        return lessThanEqual(asInt(antecedent),
179                                sub(left, domain.getLowBound()));
180                    }
181                    break;
182                case NotEqual:
183                    // Rewrite
184                    //     lone(bool, int != 888)
185                    //         where dom(int) = {-3, 888}
186                    // to
187                    //     asInt(bool) <= int - (-3)
188                    if (domain.getHighBound() == constant.intValue()) {
189                        return lessThanEqual(asInt(antecedent),
190                                sub(left, domain.getLowBound()));
191                    }
192                    // Rewrite
193                    //     lone(bool, int != -3)
194                    //         where dom(int) = {-3, 888}
195                    // to
196                    //     asInt(bool) <= 888 - int
197                    //     asInt(bool) + int <= 888
198                    if (domain.getLowBound() == constant.intValue()) {
199                        return lessThanEqual(add(asInt(antecedent), left),
200                                domain.getHighBound());
201                    }
202                    break;
203            }
204        }
205        return null;
206    }

208    /**
209     * Optimize {@code antecedent or (left `op` right)} where `op` is = or !=.
210     */
211    private static IrBoolExpr optimizeOrCompare(IrBoolExpr antecedent, IrIntExpr left, IrCompare.Op op, IrIntExpr right) {
212        IrDomain domain = left.getDomain();
213        Integer constant = IrUtil.getConstant(right);
214        if (domain.size() == 2 && constant != null) {
215            switch (op) {
216                case Equal:
217                    // Rewrite
218                    //     bool or int = 888
219                    //         where dom(int) = {-3, 888}
220                    // to
221                    //     asInt(bool) > (-3) - int
222                    //     asInt(bool) + int > (-3)
223                    if (domain.getHighBound() == constant.intValue()) {
224                        return greaterThan(add(asInt(antecedent), left),
225                                domain.getLowBound());
226                    }
227                    // Rewrite
228                    //     bool or int = -3
229                    //         where dom(int) = {-3, 888}
230                    // to
231                    //     asInt(bool) > int - 888
232                    if (domain.getLowBound() == constant.intValue()) {
233                        return greaterThan(asInt(antecedent),
234                                sub(left, domain.getHighBound()));
235                    }
236                    break;
237                case NotEqual:
238                    // Rewrite
239                    //     bool or int != 888
240                    //         where dom(int) = {-3, 888}
241                    // to
242                    //     asInt(bool) > int - 888
243                    if (domain.getHighBound() == constant.intValue()) {
244                        return greaterThan(asInt(antecedent),
245                                sub(left, domain.getHighBound()));
246                    }
247                    // Rewrite
248                    //     bool or int != -3
249                    //         where dom(int) = {-3, 888}
250                    // to
251                    //     asInt(bool) > (-3) - int
252                    //     asInt(bool) + int > (-3)
253                    if (domain.getLowBound() == constant.intValue()) {
254                        return greaterThan(add(asInt(antecedent), left),
255                                domain.getLowBound());
256                    }
257                    break;
258            }
259        }
260        return null;
261    }

263    /**
264     * Optimize {@code antecedent => (left `op` right)} where `op` is = or !=.
265     */
266    private static IrBoolExpr optimizeImplicationCompare(IrBoolExpr antecedent, IrIntExpr left, IrCompare.Op op, IrIntExpr right) {
267        IrDomain domain = left.getDomain();
268        Integer constant = IrUtil.getConstant(right);
269        if (domain.size() == 2 && constant != null) {

271                case Equal:

273                    //     bool => int = 888
274                    //         where dom(int) = {-3, 888}
275                    // to
276                    //     asInt(bool) <= int - (-3)
277                    if (domain.getHighBound() == constant.intValue()) {
278                        return lessThanEqual(asInt(antecedent),
279                                sub(left, domain.getLowBound()));
280                    }
281                    // Rewrite
282                    //     bool => int = -3
283                    //         where dom(int) = {-3, 888}
284                    // to
285                    //     asInt(bool) <= 888 - int
286                    //     asInt(bool) + int <= 888
287                    if (domain.getLowBound() == constant.intValue()) {
288                        return lessThanEqual(add(asInt(antecedent), left),
289                                domain.getHighBound());
290                    }
291                    break;
292                case NotEqual:
293                    // Rewrite
294                    //     bool => int != 888
295                    //         where dom(int) = {-3, 888}
296                    // to
297                    //     asInt(bool) <= 888 - int
298                    //     asInt(bool) + int <= 888
299                    if (domain.getHighBound() == constant.intValue()) {
300                        return lessThanEqual(add(asInt(antecedent), left),

302                    }
303                    // Rewrite
304                    //     bool => int != -3
305                    //         where dom(int) = {-3, 888}

309                        return lessThanEqual(asInt(antecedent),
310                                sub(left, domain.getLowBound()));
311                    }
312                    break;
313            }
314        }
315        return null;
316    }

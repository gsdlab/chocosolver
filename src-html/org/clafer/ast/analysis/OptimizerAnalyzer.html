001package org.clafer.ast.analysis;
002
003import java.util.ArrayList;
004import java.util.List;
005import org.clafer.ast.AstClafer;
006import org.clafer.ast.AstConcreteClafer;
007import org.clafer.ast.AstConstant;
008import org.clafer.ast.AstConstraint;
009import org.clafer.ast.AstExpr;
010import org.clafer.ast.AstExprRewriter;
011import org.clafer.ast.AstGlobal;
012import org.clafer.ast.AstJoin;
013import org.clafer.ast.AstJoinParent;
014import org.clafer.ast.AstSetExpr;
015import org.clafer.ast.AstThis;
016import static org.clafer.ast.Asts.*;
017import org.clafer.ast.Card;
018import org.clafer.common.Util;
019
020/**
021 * Optimizes the expressions inside the constraints. Assumes type checking has
022 * already passed.
023 *
024 * @author jimmy
025 */
026public class OptimizerAnalyzer extends AstExprRewriter<Analysis> implements Analyzer {
027
028    @Override
029    public Analysis analyze(Analysis analysis) {
030        List<AstConstraint> optimizedConstraint = new ArrayList<AstConstraint>();
031        for (AstConstraint constraint : analysis.getConstraints()) {
032            optimizedConstraint.add(constraint.withExpr(rewrite(constraint.getExpr(), analysis)));
033        }
034        return analysis.setConstraints(optimizedConstraint);
035    }
036
037    @Override
038    public AstExpr visit(AstJoin ast, Analysis a) {
039        AstSetExpr left = rewrite(ast.getLeft(), a);
040        if (left instanceof AstThis) {
041            if (a.getScope(a.getType(ast.getLeft())) == 1) {
042                Card childCard = a.getCard(ast.getRight());
043                if (childCard.isExact()) {
044                    return constant(ast.getRight(), Util.fromTo(0, childCard.getLow()));
045                }
046                return global(ast.getRight());
047            }
048        } else if (left instanceof AstGlobal) {
049            return global(ast.getRight());
050        } else if (left instanceof AstConstant) {
051            Card childCard = a.getCard(ast.getRight());
052            if (Format.ParentGroup.equals(a.getFormat(ast.getRight()))) {
053                AstConstant constant = (AstConstant) left;
054                assert childCard.isExact();
055                int[] childConstant = new int[constant.getValue().length * childCard.getLow()];
056                for (int i = 0; i < constant.getValue().length; i++) {
057                    for (int j = 0; j < childCard.getLow(); j++) {
058                        childConstant[i * childCard.getLow() + j] =
059                                i * constant.getValue()[i] + j;
060                    }
061                }
062                return constant(ast.getRight(), childConstant);
063            }
064            assert childCard.getLow() != a.getScope(ast.getRight()) :
065                    "Didn't run scope analysis before format analysis?";
066        }
067        return join(left, ast.getRight());
068    }
069
070    @Override
071    public AstExpr visit(AstJoinParent ast, Analysis a) {
072        AstSetExpr children = rewrite(ast.getChildren(), a);
073        if (children instanceof AstThis) {
074            AstClafer type = a.getType(ast);
075            if (a.getScope(type) == 1) {
076                return constant(type, 0);
077            }
078        } else if (children instanceof AstGlobal) {
079            AstClafer childType = a.getType(ast.getChildren());
080            if (childType instanceof AstConcreteClafer) {
081                AstConcreteClafer concreteChildType = (AstConcreteClafer) childType;
082                if (a.getCard(concreteChildType).hasLow()) {
083                    return global(a.getType(ast));
084                }
085            }
086        } else if (children instanceof AstConstant) {
087            AstConstant constant = (AstConstant) children;
088            AstClafer type = a.getType(ast);
089            if (constant.getValue().length > 0 && a.getScope(type) == 1) {
090                return constant(type, 0);
091            }
092        }
093        return joinParent(children);
094    }
095    // TDODO: rewrite for all (global)
096}

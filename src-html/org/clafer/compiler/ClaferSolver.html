001package org.clafer.compiler;
002
003import java.util.ArrayList;
004import java.util.List;
005import org.clafer.common.Check;
006import org.clafer.instance.InstanceModel;
007import solver.Solver;
008import solver.search.measure.IMeasures;
009
010/**
011 *
012 * @author jimmy
013 */
014public class ClaferSolver {
015
016    private final Solver solver;
017    private final ClaferSolutionMap solutionMap;
018    private boolean first = true;
019    private boolean more = true;
020
021    ClaferSolver(Solver solver, ClaferSolutionMap solutionMap) {
022        this.solver = Check.notNull(solver);
023        this.solutionMap = Check.notNull(solutionMap);
024    }
025
026    public IMeasures getMeasures() {
027        return solver.getMeasures();
028    }
029
030    public Solver getInternalSolver() {
031        return solver;
032    }
033
034    public ClaferSolutionMap getSolutionMap() {
035        return solutionMap;
036    }
037
038    public boolean find() {
039        if (!more) {
040            return false;
041        }
042        if (first) {
043            first = false;
044            more &= solver.findSolution();
045            return more;
046        }
047        more &= solver.nextSolution();
048        return more;
049    }
050
051    public InstanceModel instance() {
052        return solutionMap.getInstance();
053    }
054
055    public InstanceModel[] allInstances() {
056        List<InstanceModel> instances = new ArrayList<InstanceModel>();
057        while (find()) {
058            instances.add(instance());
059        }
060        return instances.toArray(new InstanceModel[instances.size()]);
061    }
062
063    @Override
064    public String toString() {
065        return solver.toString();
066    }
067}

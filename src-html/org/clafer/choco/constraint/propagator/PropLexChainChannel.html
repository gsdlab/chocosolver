003import java.util.Arrays;
004import solver.constraints.Propagator;
005import solver.constraints.PropagatorPriority;

011/**
012 * strings[i] &lt; strings[j] iff ints[i] &lt;= ints[j] strings[i] = strings[j]
013 * iff ints[i] = ints[j]
014 *
015 * 0 &lt; ints[i] &lt; |{ints}|
016 *
017 * @author jimmy
018 */
019public class PropLexChainChannel extends Propagator<IntVar> {
020
021    private final IntVar[][] strings;
022    private final IntVar[] ints;
023
024    public PropLexChainChannel(IntVar[][] strings, IntVar[] ints) {
025        super(buildArray(strings, ints), PropagatorPriority.QUADRATIC, false);
026        this.strings = strings;
027        this.ints = ints;
028    }
029
030    public static IntVar[] buildArray(IntVar[][] strings, IntVar[] ints) {
031        if (strings.length != ints.length) {
032            throw new IllegalArgumentException();
033        }
034        IntVar[] array = new IntVar[strings.length * strings[0].length + ints.length];
035        System.arraycopy(ints, 0, array, 0, ints.length);
036        int i = ints.length;
037        for (IntVar[] string : strings) {
038            System.arraycopy(string, 0, array, i, string.length);
039            i += string.length;
040        }
041        return Arrays.copyOf(array, i);
042    }
043
044    @Override
045    public int getPropagationConditions(int vIdx) {
046        return EventType.INT_ALL_MASK();
047    }
048
049    private static Ordering compareString(IntVar[] a, IntVar[] b) {
050        return compareString(a, b, 0);
051    }
052
053    private static Ordering compareString(IntVar[] a, IntVar[] b, int index) {
054        if (index == a.length) {
055            return a.length == b.length ? Ordering.EQ : Ordering.LT;
056        }
057        if (index == b.length) {
058            assert a.length != b.length;
059            return Ordering.GT;
060        }
061        Ordering ord = compare(a[index], b[index]);
062        switch (ord) {
063            case EQ:
064                return compareString(a, b, index + 1);
065            case LE:
066                switch (compareString(a, b, index + 1)) {
067                    case LT:
068                        return Ordering.LT;
069                    case LE:
070                    case EQ:
071                        return Ordering.LE;
072                    default:
073                        return Ordering.UNKNOWN;
074                }
075            case GE:
076                switch (compareString(a, b, index + 1)) {
077                    case GT:
078                        return Ordering.GT;
079                    case GE:
080                    case EQ:
081                        return Ordering.GE;
082                    default:
083                        return Ordering.UNKNOWN;
084                }
085            default:
086                return ord;
087        }
088    }
089
090    private static Ordering compare(IntVar a, IntVar b) {
091        if (a.instantiated() && b.instantiated() && a.getValue() == b.getValue()) {
092            return Ordering.EQ;
093        }
094        int aLb = a.getLB();
095        int aUb = a.getUB();
096        int bLb = b.getLB();
097        int bUb = b.getUB();
098        if (aLb > bUb) {
099            return Ordering.GT;
100        }
101        if (aLb >= bUb) {
102            return Ordering.GE;
103        }
104        if (aUb < bLb) {
105            return Ordering.LT;
106        }
107        if (aUb <= bLb) {
108            return Ordering.LE;
109        }
110        return Ordering.UNKNOWN;
111    }
112
113    private boolean lessThanString(IntVar[] a, IntVar[] b) throws ContradictionException {
114        boolean changed = false;
115        for (int x = 0; x < a.length - 1; x++) {
116            switch (compare(a[x], b[x])) {
117                case LT:
118                case LE:
119                case UNKNOWN:
120                    return changed;
121                case GT:
122                case GE:
123                    changed |= equal(a[x], b[x]);
124                    break;
125            }
126        }
127        // All equal except for the last character.
128        return lessThan(a[a.length - 1], b[a.length - 1]) || changed;
129    }
130
131    private boolean lessThan(IntVar a, IntVar b) throws ContradictionException {
132        // Don't use short-circuit or
133        return a.updateUpperBound(b.getUB() - 1, aCause)
134                | b.updateLowerBound(a.getLB() + 1, aCause);
135    }
136
137    private boolean lessThanEqualString(IntVar[] a, IntVar[] b) throws ContradictionException {
138        for (int x = 0; x < a.length; x++) {
139            Ordering charOrd = compare(a[x], b[x]);
140            if (!Ordering.EQ.equals(charOrd)) {
141                return lessThanEqual(a[x], b[x]);
142            }
143        }
144        // The two strings are equal.
145        return false;
146    }
147
148    private boolean lessThanEqual(IntVar a, IntVar b) throws ContradictionException {
149        // Don't use short-circuit or
150        return a.updateUpperBound(b.getUB(), aCause)
151                | b.updateLowerBound(a.getLB(), aCause);
152    }
153
154    private boolean greaterThanString(IntVar[] a, IntVar[] b) throws ContradictionException {
155        boolean changed = false;
156        for (int x = 0; x < a.length - 1; x++) {
157            switch (compare(a[x], b[x])) {
158                case GT:
159                case GE:
160                case UNKNOWN:
161                    return changed;
162                case LT:
163                case LE:
164                    changed |= equal(a[x], b[x]);
165                    break;
166            }
167        }
168        // All equal except for the last character.
169        return greaterThan(a[a.length - 1], b[a.length - 1]) || changed;
170    }
171
172    private boolean greaterThan(IntVar a, IntVar b) throws ContradictionException {
173        // Don't use short-circuit or
174        return a.updateLowerBound(b.getLB() + 1, aCause)
175                | b.updateUpperBound(a.getUB() - 1, aCause);
176    }
177
178    private boolean greaterThanEqualString(IntVar[] a, IntVar[] b) throws ContradictionException {
179        for (int x = 0; x < a.length; x++) {
180            Ordering charOrd = compare(a[x], b[x]);
181            if (!Ordering.EQ.equals(charOrd)) {
182                return greaterThanEqual(a[x], b[x]);
183            }
184        }
185        // The two strings are equal.
186        return false;
187    }
188
189    private boolean greaterThanEqual(IntVar a, IntVar b) throws ContradictionException {
190        // Don't use short-circuit or
191        return a.updateLowerBound(b.getLB(), aCause)
192                | b.updateUpperBound(a.getUB(), aCause);
193    }
194
195    private boolean equalString(IntVar[] a, IntVar[] b) throws ContradictionException {
196        boolean changed = false;
197        for (int i = 0; i < a.length; i++) {
198            changed |= equal(a[i], b[i]);
199        }
200        return changed;
201    }
202
203    private boolean equal(IntVar a, IntVar b) throws ContradictionException {
204        // Don't use short-circuit or
205        return a.updateLowerBound(b.getLB(), aCause)
206                | a.updateUpperBound(b.getUB(), aCause)
207                | b.updateLowerBound(a.getLB(), aCause)
208                | b.updateUpperBound(a.getUB(), aCause);
209    }
210
211    // Idempotent.
212    private boolean propagateStrings() throws ContradictionException {
213        Update operations = new Update(strings.length);
214        int eqs = 0;
215        for (int i = 0; i < strings.length; i++) {
216            boolean equivalenceClass = false;
217            for (int j = i + 1; j < strings.length; j++) {
218                Ordering ord = compareString(strings[i], strings[j]);
219                switch (ord) {
220                    case EQ:
221                        equivalenceClass = true;
222                    // fallthrough
223                    case LT:
224                    case LE:
225                    case GT:
226                    case GE:
227                        operations.add(i, j, ord);
228                        break;
229                }
230            }
231            if (equivalenceClass) {
232                eqs++;
233            }
234        }
235        boolean changed = false;
236        for (int i = 0; i < ints.length; i++) {
237            changed |= ints[i].updateUpperBound(ints.length - 1 - eqs, aCause);
238        }
239        boolean repeat;
240        do {
241            repeat = operations.doIntUpdates();
242            changed |= repeat;
243        } while (repeat);
244        return changed;
245    }
246
247    // Idempotent.
248    private boolean propagateInts() throws ContradictionException {
249        Update operations = new Update(strings.length);
250        for (int i = 0; i < ints.length; i++) {
251            for (int j = i + 1; j < ints.length; j++) {
252                Ordering ord = compare(ints[i], ints[j]);
253                if (!Ordering.UNKNOWN.equals(ord)) {
254                    operations.add(i, j, ord);
255                }
256            }
257        }
258        boolean changed = false;
259        boolean repeat;
260        do {
261            repeat = operations.doStringUpdates();
262            changed |= repeat;
263        } while (repeat);
264        return changed;
265    }
266
267    @Override
268    public void propagate(int evtmask) throws ContradictionException {
269        propagateStrings();
270        while (propagateInts() && propagateStrings());
271    }
272
273    @Override
274    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
275        forcePropagate(EventType.FULL_PROPAGATION);
276    }
277
278    @Override
279    public ESat isEntailed() {
280        for (int i = 0; i < strings.length; i++) {
281            for (int j = i + 1; j < strings.length; j++) {
282                Ordering intOrdering = compare(ints[i], ints[j]);
283                Ordering ord = Ordering.EQ;
284                for (int k = 0; Ordering.EQ.equals(ord) && k < strings[i].length; k++) {
285                    ord = compare(strings[i][k], strings[j][k]);
286                }
287                if (!intOrdering.equals(ord)
288                        && !Ordering.UNKNOWN.equals(intOrdering)
289                        && !Ordering.UNKNOWN.equals(ord)) {
290                    return ESat.FALSE;
291                }
292            }
293        }
294        return isCompletelyInstantiated() ? ESat.TRUE : ESat.UNDEFINED;
295    }
296
297    @Override
298    public String toString() {
299        return "lexChainChannel(" + Arrays.deepToString(strings) + ", " + Arrays.toString(ints) + ")";
300    }
301
302    private static enum Ordering {
303
304        LT,
305        LE,
306        GT,
307        GE,
308        EQ,
309        UNKNOWN;
310    }
311
312    private class Update {
313
314        private int[] updates;
315        private int size = 0;
316
317        Update(int capacity) {
318            updates = new int[capacity * 3];
319        }
320
321        void add(int x, int y, Ordering ord) {
322            int offset = size * 3;
323            if (offset >= updates.length) {
324                updates = Arrays.copyOf(updates, updates.length * 2);
325            }
326            updates[offset] = x;
327            updates[offset + 1] = y;
328            updates[offset + 2] = ord.ordinal();
329            size++;
330        }
331
332        boolean doStringUpdates() throws ContradictionException {
333            boolean changed = false;
334            int cap = size * 3;
335            for (int i = 0; i < cap; i += 3) {
336                int x = updates[i];
337                int y = updates[i + 1];
338                switch (updates[i + 2]) {
339                    case 0: // LT
340                        changed |= lessThanString(strings[x], strings[y]);
341                        break;
342                    case 1: // LE
343                        changed |= lessThanEqualString(strings[x], strings[y]);
344                        break;
345                    case 2: // GT
346                        changed |= greaterThanString(strings[x], strings[y]);
347                        break;
348                    case 3: // GE
349                        changed |= greaterThanEqualString(strings[x], strings[y]);
350                        break;
351                    case 4: // EQ
352                        changed |= equalString(strings[x], strings[y]);
353                        break;
354                    default:
355                        throw new IllegalStateException();
356                }
357            }
358            return changed;
359        }
360
361        boolean doIntUpdates() throws ContradictionException {
362            boolean changed = false;
363            int cap = size * 3;
364            for (int i = 0; i < cap; i += 3) {
365                int x = updates[i];
366                int y = updates[i + 1];
367                switch (updates[i + 2]) {
368                    case 0: // LT
369                        changed |= lessThan(ints[x], ints[y]);
370                        break;
371                    case 1: // LE
372                        changed |= lessThanEqual(ints[x], ints[y]);
373                        break;
374                    case 2: // GT
375                        changed |= greaterThan(ints[x], ints[y]);
376                        break;
377                    case 3: // GE
378                        changed |= greaterThanEqual(ints[x], ints[y]);
379                        break;
380                    case 4: // EQ
381                        changed |= equal(ints[x], ints[y]);
382                        break;
383                    default:
384                        throw new IllegalStateException();
385                }
386            }
387            return changed;
388        }
389    }

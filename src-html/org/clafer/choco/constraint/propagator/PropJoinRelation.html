003import gnu.trove.iterator.TIntIterator;
004import gnu.trove.list.array.TIntArrayList;
005import gnu.trove.set.hash.TIntHashSet;
006import java.util.Arrays;
007import solver.constraints.propagators.Propagator;
008import solver.constraints.propagators.PropagatorPriority;
009import solver.exception.ContradictionException;
010import solver.variables.EventType;
011import solver.variables.SetVar;
012import solver.variables.delta.monitor.SetDeltaMonitor;
013import util.ESat;
014import util.procedure.IntProcedure;

016/**
017 * <p>
018 * Join a unary relation with a binary relation. The {@code take} variable is
019 * the unary relation and the {@code children} variables are the binary
020 * relation. The {@code to} variable is the result of the join.
021 * </p>
022 * <p>
023 * Here is how the binary relation is encoded. Consider the relation:
024 * {@code (0, 1), (0, 2), (1, 3), (2, 1)}. This is encoded as 3 different
025 * {@code children} variables: {@code child0={1, 2}, child1={3}, child2={1}}.
026 * </p>
027 *
028 * @author jimmy
029 */
030public class PropJoinRelation extends Propagator<SetVar> {
031
032    private final SetVar take;
033    private final SetDeltaMonitor takeD;
034    private final SetVar[] children;
035    private final SetDeltaMonitor[] childrenD;
036    private final SetVar to;
037    private final SetDeltaMonitor toD;
038
039    public PropJoinRelation(SetVar take, SetVar[] children, SetVar to) {
040        super(buildArray(take, to, children), PropagatorPriority.LINEAR, true);
041        this.take = take;
042        this.takeD = take.monitorDelta(aCause);
043        this.children = children;
044        this.childrenD = PropUtil.monitorDeltas(children, aCause);
045        this.to = to;
046        this.toD = to.monitorDelta(aCause);
047    }
048
049    private static SetVar[] buildArray(SetVar take, SetVar to, SetVar[] children) {
050        SetVar[] array = new SetVar[children.length + 2];
051        array[0] = take;
052        array[1] = to;
053        System.arraycopy(children, 0, array, 2, children.length);
054        return array;
055    }
056
057    private boolean isTakeVar(int idx) {
058        return idx == 0;
059    }
060
061    private boolean isToVar(int idx) {
062        return idx == 1;
063    }
064
065    private boolean isChildVar(int idx) {
066        return idx >= 2;
067    }
068
069    private int getChildVarIndex(int idx) {
070        assert isChildVar(idx);
071        return idx - 2;
072    }
073
074    @Override
075    public int getPropagationConditions(int vIdx) {
076        return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
077    }
078
079    private void findMate(int toEnv) throws ContradictionException {
080        boolean inKer = to.kernelContains(toEnv);
081        int mate = -1;
082        for (int j = take.getEnvelopeFirst(); j != SetVar.END; j = take.getEnvelopeNext()) {
083            if (children[j].envelopeContains(toEnv)) {
084                // Found a second mate.
085                if (mate != -1 || !inKer) {
086                    mate = -2;
087                    break;
088                }
089                mate = j;
090            }
091        }
092        if (mate == -1) {
093            // No mates.
094            to.removeFromEnvelope(toEnv, aCause);
095        } else if (mate != -2 && inKer) {
096            // One mate.
097            take.addToKernel(mate, aCause);
098            PropUtil.kerSubsetKer(children[mate], to, aCause);
099            PropUtil.envSubsetEnv(children[mate], to, aCause);
100            children[mate].addToKernel(toEnv, aCause);
101        }
102    }
103
104    @Override
105    public void propagate(int evtmask) throws ContradictionException {
106        // Pick to and prune child
107        for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
108            PropUtil.kerSubsetKer(children[i], to, aCause);
109            PropUtil.envSubsetEnv(children[i], to, aCause);
110        }
111
112        // Prune take
113        for (int i = take.getEnvelopeFirst(); i != SetVar.END; i = take.getEnvelopeNext()) {
114            if (!PropUtil.isKerSubsetEnv(children[i], to)) {
115                take.removeFromEnvelope(i, aCause);
116            }
117        }
118
119        // Pick take, pick child, pick take, prune to
120        for (int i = to.getEnvelopeFirst(); i != SetVar.END; i = to.getEnvelopeNext()) {
121            findMate(i);
122        }
123    }
124
125    @Override
126    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
127        if (isTakeVar(idxVarInProp)) {
128            takeD.freeze();
129            takeD.forEach(pruneToOnTakeEnv, EventType.REMOVE_FROM_ENVELOPE);
130            takeD.forEach(pickToAndPruneChildOnTakeKer, EventType.ADD_TO_KER);
131            takeD.unfreeze();
132        } else if (isToVar(idxVarInProp)) {
133            toD.freeze();
134            toD.forEach(pruneChildOnToEnv, EventType.REMOVE_FROM_ENVELOPE);
135            toD.forEach(pickTakeOnToKer, EventType.ADD_TO_KER);
136            toD.unfreeze();
137            if ((EventType.REMOVE_FROM_ENVELOPE.mask & mask) != 0) {
138                TIntArrayList removed = null;
139                for (int i = take.getEnvelopeFirst(); i != SetVar.END; i = take.getEnvelopeNext()) {
140                    if (!PropUtil.isKerSubsetEnv(children[i], to)) {
141                        take.removeFromEnvelope(i, aCause);
142                        // Cannot call findMate here because we inside iterating take env.
143                        // Queue up the even to do later.
144                        if (removed == null) {
145                            removed = new TIntArrayList(1);
146                        }
147                        removed.add(i);
148                    }
149                }
150                if (removed != null) {
151                    TIntIterator it = removed.iterator();
152                    while (it.hasNext()) {
153                        SetVar child = children[it.next()];
154                        for (int i = child.getEnvelopeFirst(); i != SetVar.END; i = child.getEnvelopeNext()) {
155                            if (to.envelopeContains(i)) {
156                                findMate(i);
157                            }
158                        }
159                    }
160                }
161            }
162        } else {
163            assert isChildVar(idxVarInProp);
164            final int id = getChildVarIndex(idxVarInProp);
165            childrenD[id].freeze();
166            childrenD[id].forEach(pruneToOnChildEnv, EventType.REMOVE_FROM_ENVELOPE);
167            if (take.envelopeContains(id)) {
168                if (take.kernelContains(id)) {
169                    childrenD[id].forEach(pickToOnChildKer, EventType.ADD_TO_KER);
170                } else {
171                    IntProcedure pruneTakeOnChildKer = new IntProcedure() {
172                        @Override
173                        public void execute(int childKer) throws ContradictionException {
174                            if (!to.envelopeContains(childKer)) {
175                                take.removeFromEnvelope(id, aCause);
176                                for (int i = children[id].getEnvelopeFirst(); i != SetVar.END; i = children[id].getEnvelopeNext()) {
177                                    if (to.envelopeContains(i)) {
178                                        findMate(i);
179                                    }
180                                }
181                            }
182                        }
183                    };
184                    childrenD[id].forEach(pruneTakeOnChildKer, EventType.ADD_TO_KER);
185                }
186            }
187            childrenD[id].unfreeze();
188        }
189    }
190    private final IntProcedure pruneToOnTakeEnv = new IntProcedure() {
191        @Override
192        public void execute(int takeEnv) throws ContradictionException {
193            assert !take.envelopeContains(takeEnv);
194
195            for (int i = children[takeEnv].getEnvelopeFirst(); i != SetVar.END; i = children[takeEnv].getEnvelopeNext()) {
196                if (to.envelopeContains(i)) {
197                    findMate(i);
198                }
199            }
200        }
201    };
202    private final IntProcedure pickToAndPruneChildOnTakeKer = new IntProcedure() {
203        @Override
204        public void execute(int takeKer) throws ContradictionException {
205            assert take.kernelContains(takeKer);
206
207            SetVar child = children[takeKer];
208            PropUtil.kerSubsetKer(child, to, aCause);
209            PropUtil.envSubsetEnv(child, to, aCause);
210        }
211    };
212    private final IntProcedure pruneToOnChildEnv = new IntProcedure() {
213        @Override
214        public void execute(int childEnv) throws ContradictionException {
215            // Note the the child may no longer be in take, but still need to find mate.
216            // For example:
217            //     take = {0,1,2}, child0 = {0}, child1 = {1}, child2 = {2}, to = {0,1,2}
218            //   remove 2 from take and 2 from child2
219            //     take = {0,1}, child0 = {0}, child1 = {1}, child2 = {}, to = {0,1,2}
220            // Need to find mate for 2 on child2 or else to will keep 2, and break
221            // idempotency.
222            findMate(childEnv);
223        }
224    };
225    private final IntProcedure pickToOnChildKer = new IntProcedure() {
226        @Override
227        public void execute(int childKer) throws ContradictionException {
228            // assert id in ker(take)
229            to.addToKernel(childKer, aCause);
230        }
231    };
232    private final IntProcedure pruneChildOnToEnv = new IntProcedure() {
233        @Override
234        public void execute(int toEnv) throws ContradictionException {
235            assert !to.envelopeContains(toEnv);
236
237            for (int takeKer = take.getKernelFirst(); takeKer != SetVar.END; takeKer = take.getKernelNext()) {
238                children[takeKer].removeFromEnvelope(toEnv, aCause);
239            }
240        }
241    };
242    private final IntProcedure pickTakeOnToKer = new IntProcedure() {
243        @Override
244        public void execute(int toVal) throws ContradictionException {
245            assert to.kernelContains(toVal);
246            findMate(toVal);
247        }
248    };
249
250    @Override
251    public ESat isEntailed() {
252        for (int i = take.getKernelFirst(); i != SetVar.END; i = take.getKernelNext()) {
253            for (int j = children[i].getKernelFirst(); j != SetVar.END; j = children[i].getKernelNext()) {
254                if (!to.envelopeContains(j)) {
255                    return ESat.FALSE;
256                }
257            }
258        }
259        int count = 0;
260        for (int i = take.getEnvelopeFirst(); i != SetVar.END; i = take.getEnvelopeNext()) {
261            count += children[i].getEnvelopeSize();
262        }
263        if (count < to.getKernelSize()) {
264            return ESat.FALSE;
265        }
266        if (!take.instantiated() || !to.instantiated()) {
267            return ESat.UNDEFINED;
268        }
269        for (SetVar child : children) {
270            if (!child.instantiated()) {
271                return ESat.UNDEFINED;
272            }
273        }
274        return ESat.TRUE;
275    }
276
277    @Override
278    public String toString() {
279        return "joinRelation(" + take + ", " + Arrays.toString(children) + ", " + to + ")";
280    }

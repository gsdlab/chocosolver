053    /**
054     * Checks if it is possible for an integer variable to instantiate to a
055     * value in the set variable. Assumes the set variables envelope is sorted.
056     *
057     * @param ivar the integer variable
058     * @param svar the set variable whose envelope is sorted
059     * @return {@code true} if and only if {@code (dom(ivar) intersect env(svar)) !=
060     *         empty set}, {@code false} otherwise
061     */
062    public static boolean domainIntersectEnv(IntVar ivar, SetVar svar) {
063        if (ivar.getDomainSize() < svar.getEnvelopeSize()) {
064            int i = ivar.getLB();
065            int envFirst = svar.getEnvelopeFirst();
066            if (envFirst >= i) {
067                if (envFirst == i || ivar.contains(envFirst)) {
068                    return true;
069                }
070                i = ivar.nextValue(envFirst);
071            }
072            int ub = ivar.getUB();
073            for (; i <= ub; i = ivar.nextValue(i)) {
074                if (svar.envelopeContains(i)) {
075                    return true;
076                }
077            }
078        } else {
079            for (int i = svar.getEnvelopeFirst(); i != SetVar.END;) {
080                int next = ivar.nextValue(i - 1);
081                while (i < next && i != SetVar.END) {
082                    i = svar.getEnvelopeNext();
083                }
084                if (i == next) {
085                    return true;
086                }
087            }
088        }
089        return false;
090    }

092    /**
093     * Checks if it is guaranteed that an integer variable instantiates to a
094     * value in a collection.
095     *
096     * @param ivar the integer variable
097     * @param collection the collection
098     * @return {@code true} if and only if {@code dom(ivar) subsetof collection},
099     *         {@code false} otherwise
100     */
101    public static boolean isDomainSubsetOf(IntVar ivar, TIntCollection collection) {
102        if (ivar.getDomainSize() > collection.size()) {
103            return false;
104        }
105        int ub = ivar.getUB();
106        for (int i = ivar.getLB(); i <= ub; i = ivar.nextValue(i)) {
107            if (!collection.contains(i)) {
108                return false;
109            }
110        }
111        return true;
112    }

114    /**
115     * Checks if it is guaranteed that an integer variable instantiates to a
116     * value in the set variable.
117     *
118     * @param ivar the integer variable
119     * @param svar the set variable
120     * @return {@code true} if and only if {@code dom(ivar) subsetof env(svar)},
121     *         {@code false} otherwise
122     */
123    public static boolean isDomainSubsetEnv(IntVar ivar, SetVar svar) {
124        if (ivar.getDomainSize() < svar.getKernelSize()) {
125            int ub = ivar.getUB();
126            for (int i = Math.max(ivar.getLB(), svar.getKernelFirst()); i <= ub; i = ivar.nextValue(i)) {
127                if (!svar.kernelContains(i)) {
128                    return false;
129                }
130            }
131        } else {
132            for (int i = svar.getKernelFirst(); i != SetVar.END;) {
133                int next = ivar.nextValue(i - 1);
134                while (i < next && i != SetVar.END) {
135                    i = svar.getKernelNext();
136                }
137                if (i == next) {
138                    return false;
139                }
140            }
141        }
142        return true;
143    }

145    /**
146     * Checks if a collection is contained entirely in a set variable's
147     * envelope.
148     *
149     * @param sub the subset
150     * @param sup the superset
151     * @return {@code true} if and only if {@code sub subsetof env(sup)},
152     *         {@code false} otherwise
153     */
154    public static boolean isSubsetOfEnv(TIntCollection sub, SetVar sup) {
155        TIntIterator iter = sub.iterator();
156        while (iter.hasNext()) {
157            if (!sup.envelopeContains(iter.next())) {
158                return false;
159            }
160        }
161        return true;
162    }

317    /**
318     * Removes every element in the integer variable's domain that is not in the
319     * envelope of the set variable.
320     *
321     * @param sub the subset
322     * @param sup the superset
323     * @param propagator the propagator
324     * @throws ContradictionException
325     */
326    public static void intSubsetEnv(IntVar sub, SetVar sup, ICause propagator) throws ContradictionException {
327        int left = Integer.MIN_VALUE;
328        int right = left;
329        int ub = sub.getUB();
330        for (int val = sub.getLB(); val <= ub; val = sub.nextValue(val)) {
331            if (!sup.envelopeContains(val)) {
332                if (val == right + 1) {
333                    right = val;
334                } else {
335                    sub.removeInterval(left, right, propagator);
336                    left = val;
337                    right = val;
338                }
339            }
340        }
341        sub.removeInterval(left, right, propagator);
342    }

408    /**
409     * Enumerate the envelope of a set variable.
410     *
411     * @param set the set variable
412     * @return {@code env(set)}
413     */
414    public static int[] iterateEnv(SetVar set) {
415        int[] iterate = new int[set.getEnvelopeSize()];
416        int count = 0;
417        for (int i = set.getEnvelopeFirst(); i != SetVar.END; i = set.getEnvelopeNext()) {
418            iterate[count++] = i;
419        }
420        assert count == iterate.length;
421        return iterate;
422    }

464    /**
465     * Returns the integer variables' values. Assumes the variables are
466     * instantiated.
467     *
468     * @param vars the variables
469     * @return the variables' values
470     */
471    public static int[] getValues(IntVar[] vars) {
472        int[] values = new int[vars.length];
473        for (int i = 0; i < values.length; i++) {
474            assert vars[i].instantiated();
475            values[i] = vars[i].getValue();
476        }
477        return values;
478    }

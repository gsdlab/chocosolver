001package org.clafer.choco.constraint.propagator;
002
003import org.clafer.common.Util;
004import solver.constraints.propagators.Propagator;
005import solver.constraints.propagators.PropagatorPriority;
006import solver.exception.ContradictionException;
007import solver.variables.EventType;
008import solver.variables.SetVar;
009import solver.variables.delta.monitor.SetDeltaMonitor;
010import util.ESat;
011import util.procedure.IntProcedure;
012
013/**
014 * Idempotent version of the one provided by the Choco library.
015 *
016 * @author jimmy
017 */
018public class PropSetUnion extends Propagator<SetVar> {
019
020    private final SetVar[] sets;
021    private final SetDeltaMonitor[] setsD;
022    private final SetVar union;
023    private final SetDeltaMonitor unionD;
024
025    public PropSetUnion(SetVar[] sets, SetVar union) {
026        super(Util.cons(union, sets), PropagatorPriority.LINEAR, true);
027        this.sets = sets;
028        this.setsD = PropUtil.monitorDeltas(sets, aCause);
029        this.union = union;
030        this.unionD = union.monitorDelta(aCause);
031    }
032
033    private boolean isSetVar(int idx) {
034        return idx >= 1;
035    }
036
037    private int getSetVarIndex(int idx) {
038        assert isSetVar(idx);
039        return idx - 1;
040    }
041
042    private boolean isUnionVar(int idx) {
043        return idx == 0;
044    }
045
046    @Override
047    public int getPropagationConditions(int vIdx) {
048        return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
049    }
050
051    private void findMate(int unionEnv) throws ContradictionException {
052        boolean inKer = union.kernelContains(unionEnv);
053        int mate = -1;
054        for (int j = 0; j < sets.length; j++) {
055            if (sets[j].envelopeContains(unionEnv)) {
056                // Found a second mate.
057                if (mate != -1 || !inKer) {
058                    mate = -2;
059                    break;
060                }
061                mate = j;
062            }
063        }
064        if (mate == -1) {
065            // No mates.
066            union.removeFromEnvelope(unionEnv, aCause);
067        } else if (mate != -2 && inKer) {
068            // One mate.
069            sets[mate].addToKernel(unionEnv, aCause);
070        }
071    }
072
073    @Override
074    public void propagate(int evtmask) throws ContradictionException {
075        for (SetVar set : sets) {
076            PropUtil.envSubsetEnv(set, union, aCause);
077            PropUtil.kerSubsetKer(set, union, aCause);
078        }
079        PropUtil.envSubsetEnvs(union, sets, aCause);
080        for (int i = union.getEnvelopeFirst(); i != SetVar.END; i = union.getEnvelopeNext()) {
081            findMate(i);
082        }
083    }
084
085    @Override
086    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
087        if (isSetVar(idxVarInProp)) {
088            int id = getSetVarIndex(idxVarInProp);
089            setsD[id].freeze();
090            setsD[id].forEach(pruneUnionOnSetEnv, EventType.REMOVE_FROM_ENVELOPE);
091            setsD[id].forEach(pickUnionOnSetKer, EventType.ADD_TO_KER);
092            setsD[id].unfreeze();
093        } else {
094            assert isUnionVar(idxVarInProp);
095            unionD.freeze();
096            unionD.forEach(pruneSetOnUnionEnv, EventType.REMOVE_FROM_ENVELOPE);
097            unionD.forEach(pickSetOnUnionKer, EventType.ADD_TO_KER);
098            unionD.unfreeze();
099        }
100    }
101    private final IntProcedure pruneUnionOnSetEnv = new IntProcedure() {
102        @Override
103        public void execute(int setEnv) throws ContradictionException {
104            findMate(setEnv);
105        }
106    };
107    private final IntProcedure pickUnionOnSetKer = new IntProcedure() {
108        @Override
109        public void execute(int setKer) throws ContradictionException {
110            union.addToKernel(setKer, aCause);
111        }
112    };
113    private final IntProcedure pruneSetOnUnionEnv = new IntProcedure() {
114        @Override
115        public void execute(int unionEnv) throws ContradictionException {
116            for (SetVar set : sets) {
117                set.removeFromEnvelope(unionEnv, aCause);
118            }
119        }
120    };
121    private final IntProcedure pickSetOnUnionKer = new IntProcedure() {
122        @Override
123        public void execute(int unionKer) throws ContradictionException {
124            findMate(unionKer);
125        }
126    };
127
128    @Override
129    public ESat isEntailed() {
130        boolean allInstantiated = true;
131        for (SetVar set : sets) {
132            if (!PropUtil.isKerSubsetEnv(set, union)) {
133                return ESat.FALSE;
134            }
135            allInstantiated &= set.instantiated();
136        }
137        if (!PropUtil.isEnvSubsetEnvs(union, sets)) {
138            return ESat.FALSE;
139        }
140        allInstantiated &= union.instantiated();
141        return allInstantiated ? ESat.TRUE : ESat.UNDEFINED;
142    }
143
144    @Override
145    public String toString() {
146        return Util.intercalate(" âˆª ", sets) + " = " + union;
147    }
148}

001package org.clafer.choco.constraint.propagator;
002
003import org.clafer.collection.FixedCapacityIntSet;
004import org.clafer.common.Util;
005import solver.constraints.propagators.Propagator;
006import solver.constraints.propagators.PropagatorPriority;
007import solver.exception.ContradictionException;
008import solver.variables.EventType;
009import solver.variables.IntVar;
010import solver.variables.SetVar;
011import solver.variables.Variable;
012import solver.variables.delta.IIntDeltaMonitor;
013import solver.variables.delta.monitor.SetDeltaMonitor;
014import util.ESat;
015import util.procedure.IntProcedure;
016
017/**
018 *
019 * @author jimmy
020 */
021public class PropArrayToSet extends Propagator<Variable> {
022
023    private final IntVar[] as;
024    private final IIntDeltaMonitor[] asD;
025    private final SetVar s;
026    private final SetDeltaMonitor sD;
027
028    public PropArrayToSet(IntVar[] as, SetVar s) {
029        super(buildArray(as, s), PropagatorPriority.BINARY, true);
030        if (as.length == 0) {
031            throw new IllegalArgumentException();
032        }
033        this.as = as;
034        this.asD = PropUtil.monitorDeltas(as, aCause);
035        this.s = s;
036        this.sD = s.monitorDelta(aCause);
037    }
038
039    private static Variable[] buildArray(IntVar[] as, SetVar s) {
040        Variable[] array = new Variable[as.length + 1];
041        array[0] = s;
042        System.arraycopy(as, 0, array, 1, as.length);
043        return array;
044    }
045
046    private boolean isAVar(int idx) {
047        return idx > 0;
048    }
049
050    private int getAVarIndex(int idx) {
051        assert isAVar(idx);
052        return idx - 1;
053    }
054
055    private boolean isSVar(int idx) {
056        return idx == 0;
057    }
058
059    @Override
060    public int getPropagationConditions(int vIdx) {
061        if (isAVar(vIdx)) {
062            return EventType.INT_ALL_MASK();
063        }
064        assert isSVar(vIdx);
065        return EventType.REMOVE_FROM_ENVELOPE.mask + EventType.ADD_TO_KER.mask;
066    }
067
068    private void checkKerSize() throws ContradictionException {
069        if (s.getKernelSize() > as.length) {
070            contradiction(s, s + " is too large");
071        }
072        if (s.getKernelSize() == as.length) {
073            if (!s.instantiated()) {
074                s.instantiateTo(PropUtil.iterateKer(s), aCause);
075                PropUtil.intsSubsetEnv(as, s, aCause);
076            }
077        }
078    }
079
080    @Override
081    public void propagate(int evtmask) throws ContradictionException {
082        // Prune as
083        PropUtil.intsSubsetEnv(as, s, aCause);
084        // Prune s
085        PropUtil.envSubsetInts(s, as, aCause);
086        // Pick s
087        for (IntVar a : as) {
088            if (a.instantiated()) {
089                s.addToKernel(a.getValue(), aCause);
090            }
091        }
092        checkKerSize();
093    }
094
095    @Override
096    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
097        if (isSVar(idxVarInProp)) {
098            sD.freeze();
099            sD.forEach(pruneAOnSEnv, EventType.REMOVE_FROM_ENVELOPE);
100            sD.unfreeze();
101            checkKerSize();
102        } else {
103            assert isAVar(idxVarInProp);
104
105            int id = getAVarIndex(idxVarInProp);
106            if (EventType.isRemove(mask)
107                    || (EventType.isInclow(mask) && as[id].getLB() > s.getEnvelopeFirst())
108                    || EventType.isDecupp(mask)) {
109                asD[id].freeze();
110                asD[id].forEach(pruneSOnARem, EventType.REMOVE);
111                asD[id].unfreeze();
112            }
113            if (as[id].instantiated()) {
114                s.addToKernel(as[id].getValue(), aCause);
115                checkKerSize();
116            }
117        }
118    }
119    private final IntProcedure pruneAOnSEnv = new IntProcedure() {
120        @Override
121        public void execute(int sEnv) throws ContradictionException {
122            for (IntVar a : as) {
123                if (a.removeValue(sEnv, aCause) && a.instantiated()) {
124                    s.addToKernel(a.getValue(), aCause);
125                }
126            }
127        }
128    };
129    private final IntProcedure pruneSOnARem = new IntProcedure() {
130        @Override
131        public void execute(int aRem) throws ContradictionException {
132            if (s.envelopeContains(aRem)) {
133                for (IntVar a : as) {
134                    if (a.contains(aRem)) {
135                        return;
136                    }
137                }
138                s.removeFromEnvelope(aRem, aCause);
139            }
140        }
141    };
142
143    @Override
144    public ESat isEntailed() {
145        if (s.getKernelSize() > as.length) {
146            return ESat.FALSE;
147        }
148        boolean tsInstantiated = true;
149        FixedCapacityIntSet values = new FixedCapacityIntSet(as.length);
150        for (IntVar a : as) {
151            if (!PropUtil.domainIntersectEnv(a, s)) {
152                return ESat.FALSE;
153            }
154            if (a.instantiated()) {
155                values.add(a.getValue());
156            } else {
157                tsInstantiated = false;
158            }
159
160        }
161        if (!PropUtil.isSubsetOfEnv(values, s)) {
162            return ESat.FALSE;
163        }
164        if (tsInstantiated) {
165            return PropUtil.isKerSubsetOf(s, values)
166                    ? (s.instantiated() ? ESat.TRUE : ESat.UNDEFINED)
167                    : ESat.FALSE;
168        }
169        return ESat.UNDEFINED;
170    }
171
172    @Override
173    public String toString() {
174        return "{" + Util.commaSeparate(as) + "} = " + s;
175    }
176}

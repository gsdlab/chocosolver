001package org.clafer.choco.constraint.propagator;
002
003import java.util.Arrays;
004import solver.constraints.Propagator;
005import solver.constraints.PropagatorPriority;
006import solver.exception.ContradictionException;
007import solver.variables.EventType;
008import solver.variables.IntVar;
009import solver.variables.SetVar;
010import solver.variables.Variable;
011import util.ESat;
012
013/**
014 * result = [string !! i | i <- set].
015 *
016 * Pads result with extra negative one if necessary.
017 *
018 * @author jimmy
019 */
020public class PropFilterString extends Propagator<Variable> {
021
022    private final SetVar set;
023    private final IntVar setCard;
024    private final int offset;
025    // Sorted in decreasing order. Non-negatives.
026    private final IntVar[] string;
027    private final IntVar[] result;
028
029    public PropFilterString(SetVar set, IntVar setCard, int offset, IntVar[] string, IntVar[] result) {
030        super(buildArray(set, setCard, string, result), PropagatorPriority.QUADRATIC, false);
031        this.set = set;
032        this.setCard = setCard;
033        this.offset = offset;
034        this.string = string;
035        this.result = result;
036    }
037
038    public static Variable[] buildArray(SetVar set, IntVar setCard, IntVar[] string, IntVar[] result) {
039        Variable[] array = new Variable[2 + string.length + result.length];
040        array[0] = set;
041        array[1] = setCard;
042        System.arraycopy(string, 0, array, 2, string.length);
043        System.arraycopy(result, 0, array, 2 + string.length, result.length);
044        return array;
045    }
046
047    private boolean isSetVar(int idx) {
048        return idx == 0;
049    }
050
051    private boolean isSetCardVar(int idx) {
052        return idx == 1;
053    }
054
055    private boolean isStringVar(int idx) {
056        return idx >= 1 && idx < string.length + 2;
057    }
058
059    private int getStringVarIndex(int idx) {
060        assert isStringVar(idx);
061        return idx - 2;
062    }
063
064    private boolean isResultVar(int idx) {
065        return idx >= string.length + 2;
066    }
067
068    private int getResultVarIndex(int idx) {
069        assert isResultVar(idx);
070        return idx - 2 - string.length;
071    }
072
073    @Override
074    public boolean advise(int idxVarInProp, int mask) {
075        if (isStringVar(idxVarInProp)) {
076            return set.envelopeContains(getStringVarIndex(idxVarInProp) + offset);
077        }
078        return super.advise(idxVarInProp, mask);
079    }
080
081    @Override
082    public int getPropagationConditions(int vIdx) {
083        if (isSetVar(vIdx)) {
084            return EventType.ADD_TO_KER.mask + EventType.REMOVE_FROM_ENVELOPE.mask;
085        }
086        if (isSetCardVar(vIdx)) {
087            return EventType.BOUND.mask + EventType.INSTANTIATE.mask;
088        }
089        return EventType.INT_ALL_MASK();
090    }
091
092    private boolean subset(IntVar sub, IntVar[] sups, int from, int to) throws ContradictionException {
093        boolean changed = false;
094        int ub = sub.getUB();
095        for (int val = sub.getLB(); val <= ub; val = sub.nextValue(val)) {
096            boolean found = false;
097            for (int i = from; i < to; i++) {
098                if (sups[i].contains(val)) {
099                    found = true;
100                    break;
101                }
102            }
103            if (!found) {
104                changed |= sub.removeValue(val, aCause);
105            }
106        }
107        return changed;
108    }
109
110    private boolean subset(IntVar sub, IntVar[] sups, int[] indices, int low, int to) throws ContradictionException {
111        boolean changed = false;
112        int ub = sub.getUB();
113        for (int val = sub.getLB(); val <= ub; val = sub.nextValue(val)) {
114            boolean found = false;
115            for (int i = low; i < to; i++) {
116                if (sups[indices[i]].contains(val)) {
117                    found = true;
118                    break;
119                }
120            }
121            if (!found) {
122                changed |= sub.removeValue(val, aCause);
123            }
124        }
125        return changed;
126    }
127
128    private boolean subsetOrNegativeOne(IntVar sub, IntVar[] sups, int[] indices, int from, int to) throws ContradictionException {
129        boolean changed = false;
130        int ub = sub.getUB();
131        for (int val = sub.getLB(); val <= ub; val = sub.nextValue(val)) {
132            if (val != -1) {
133                boolean found = false;
134                for (int i = from; i < to; i++) {
135                    assert i < indices.length;
136                    if (sups[indices[i]].contains(val)) {
137                        found = true;
138                        break;
139                    }
140                }
141                if (!found) {
142                    changed |= sub.removeValue(val, aCause);
143                }
144            }
145        }
146        return changed;
147    }
148
149    @Override
150    public void propagate(int evtmask) throws ContradictionException {
151        if (set.getKernelSize() > result.length) {
152            contradiction(set, "Too many in kernel");
153        }
154
155        int[] env = new int[set.getEnvelopeSize()];
156        int[] kerIndices = new int[set.getKernelSize()];
157
158        boolean changed;
159        do {
160            changed = false;
161            // The number of ker elements seen.
162            int kerIndex = 0;
163            // The number of env elements seen.
164            int envIndex = 0;
165            for (int i = set.getEnvelopeFirst(); i != SetVar.END; i = set.getEnvelopeNext()) {
166                int x = i - offset;
167                if (x < 0 || x >= string.length) {
168                    set.removeFromEnvelope(i, aCause);
169                } else {
170                    env[envIndex] = x;
171                    if (set.kernelContains(i)) {
172                        kerIndices[kerIndex] = envIndex;
173                        changed |= subset(string[x], result, kerIndex, Math.min(envIndex + 1, result.length));
174                        envIndex++;
175                        kerIndex++;
176                    } else {
177                        boolean found = false;
178                        for (int j = kerIndex; j < result.length && j <= envIndex; j++) {
179                            if (PropUtil.isDomIntersectDom(string[x], result[j])) {
180                                found = true;
181                                break;
182                            }
183                        }
184                        if (!found) {
185                            changed |= set.removeFromEnvelope(i, aCause);
186                        } else {
187                            envIndex++;
188                        }
189                    }
190                }
191            }
192            assert envIndex <= env.length;
193            assert kerIndex == kerIndices.length;
194
195            int lb = setCard.getLB();
196            int ub = setCard.getUB();
197
198            for (; lb < result.length && !result[lb].contains(-1); lb++) {
199            }
200            changed |= setCard.updateLowerBound(lb, aCause);
201            for (int i = 0; i < result.length; i++) {
202                if (i < kerIndices.length) {
203                    changed |= subset(result[i], string, env, i, kerIndices[i] + 1);
204                } else if (i < lb) {
205                    changed |= subset(result[i], string, env, i, envIndex);
206                } else if (i > ub) {
207                    changed |= result[i].instantiateTo(-1, aCause);
208                } else {
209                    changed |= subsetOrNegativeOne(result[i], string, env, i, envIndex);
210                }
211            }
212        } while (changed);
213    }
214
215    @Override
216    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
217        forcePropagate(EventType.FULL_PROPAGATION);
218    }
219
220    @Override
221    public ESat isEntailed() {
222        int index = 0;
223        for (int i = set.getEnvelopeFirst(); i != SetVar.END && set.kernelContains(i); i = set.getEnvelopeNext(), index++) {
224            if (index >= result.length) {
225                return ESat.FALSE;
226            }
227            int x = i - offset;
228            if (x < 0 || x >= string.length) {
229                return ESat.FALSE;
230            }
231            if (string[x].instantiated() && result[index].instantiated()
232                    && string[x].getValue() != result[index].getValue()) {
233                return ESat.FALSE;
234            }
235        }
236        for (int i = set.getEnvelopeSize(); i < result.length; i++) {
237            if (!result[i].contains(-1)) {
238                return ESat.FALSE;
239            }
240        }
241        return isCompletelyInstantiated() ? ESat.TRUE : ESat.UNDEFINED;
242    }
243
244    @Override
245    public String toString() {
246        return "filter(" + set + " >> " + offset + ", " + Arrays.toString(string) + ", " + Arrays.toString(result) + ")";
247    }

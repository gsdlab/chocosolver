001package org.clafer.choco.constraint.propagator;
002
003import solver.constraints.Propagator;
004import solver.constraints.PropagatorPriority;
005import solver.exception.ContradictionException;
006import solver.variables.BoolVar;
007import solver.variables.EventType;
008import solver.variables.IntVar;
009import util.ESat;
010import static util.ESat.FALSE;
011import static util.ESat.TRUE;
012
013/**
014 *
015 * @author jimmy
016 */
017public class PropReifyNotEqualXC  extends Propagator<IntVar> {
018
019    private final BoolVar reify;
020    private final IntVar x;
021    private final int c;
022
023    public PropReifyNotEqualXC(BoolVar reify, IntVar x, int c) {
024        super(new IntVar[]{reify, x}, PropagatorPriority.UNARY, true);
025        this.reify = reify;
026        this.x = x;
027        this.c = c;
028    }
029
030    private boolean isReifyVar(int idx) {
031        return idx == 0;
032    }
033
034    private boolean isXVar(int idx) {
035        return idx == 1;
036    }
037
038    @Override
039    public int getPropagationConditions(int vIdx) {
040        return EventType.INT_ALL_MASK();
041    }
042
043    private void propagateReifyVar() throws ContradictionException {
044        assert reify.instantiated();
045        if (reify.getValue() == 0) {
046            x.instantiateTo(c, aCause);
047        } else {
048            x.removeValue(c, aCause);
049        }
050        setPassive();
051    }
052
053    private void propagateXVar() throws ContradictionException {
054        if (x.contains(c)) {
055            if (x.instantiated()) {
056                reify.setToFalse(aCause);
057                setPassive();
058            }
059        } else {
060            reify.setToTrue(aCause);
061            setPassive();
062        }
063    }
064
065    @Override
066    public void propagate(int evtmask) throws ContradictionException {
067        if (reify.instantiated()) {
068            propagateReifyVar();
069        } else {
070            propagateXVar();
071        }
072    }
073
074    @Override
075    public void propagate(int idxVarInProp, int mask) throws ContradictionException {
076        if (isReifyVar(idxVarInProp)) {
077            propagateReifyVar();
078        } else {
079            assert isXVar(idxVarInProp);
080            propagateXVar();
081        }
082    }
083
084    @Override
085    public ESat isEntailed() {
086        switch (reify.getBooleanValue()) {
087            case TRUE:
088                if (!x.contains(c)) {
089                    return ESat.TRUE;
090                }
091                if (x.instantiated()) {
092                    return ESat.FALSE;
093                }
094                break;
095            case FALSE:
096                if (!x.contains(c)) {
097                    return ESat.FALSE;
098                }
099                if (x.instantiated()) {
100                    return ESat.TRUE;
101                }
102                break;
103        }
104        return ESat.UNDEFINED;
105    }
106
107    @Override
108    public String toString() {
109        return reify + " => (" + x + " != " + c + ")";
110    }
111}

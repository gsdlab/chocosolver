051    public static Constraint arrayToSet(IntVar[] ivars, SetVar svar) {
052        Variable[] vars = new Variable[ivars.length + 1];
053        System.arraycopy(ivars, 0, vars, 0, ivars.length);
054        vars[ivars.length] = svar;
055        Constraint constraint = new Constraint(vars, svar.getSolver());
056        constraint.setPropagators(new PropArrayToSet(ivars, svar));
057        return constraint;
058    }

084    public static Constraint joinRelation(SetVar take, SetVar[] children, SetVar to) {
085        SetVar[] vars = new SetVar[children.length + 2];
086        vars[0] = take;
087        vars[1] = to;
088        System.arraycopy(children, 0, vars, 2, children.length);
089        Constraint<SetVar, PropJoinRelation> constraint = new Constraint<SetVar, PropJoinRelation>(vars, take.getSolver());
090        constraint.setPropagators(new PropJoinRelation(take, children, to));
091        return constraint;
092    }

122    /**
123     * <p>
124     * Sum the set with a maximum cardinality. More efficient than the standard
125     * operation in the Choco library when n is relatively small.
126     * </p>
127     * <p>
128     * For example:
129     * <pre>
130     *   Animal 2
131     *     Age -> integer
132     *   [Animal.Age = 1000]
133     * </pre>
134     * </p>
135     * <p> {@code Animal.Age} is a set with a very large envelope. However, due
136     * to static analysis of the model, it is easy to see that the cardinality
137     * can be no larger than 2, hence n=2. Once the first integer x is selected
138     * for the set, the second integer 1000 - x is already determined due to
139     * n=2. Since the Choco library's setSum constraint is not given n, it
140     * cannot make this deduction.
141     * </p>
142     *
143     * @param set the set of integers
144     * @param sum the sum of the set
145     * @param n the maximum cardinality of the set
146     * @return a constraint where |set| &le n and sum = Î£ set
147     */
148    public static Constraint setSumN(SetVar set, IntVar sum, int n) {
149        Constraint constraint = new Constraint(new Variable[]{set, sum}, set.getSolver());
150        constraint.setPropagators(new PropSetSumN(set, sum, n));
151        return constraint;
152    }
